#!/bin/sh

update_crontab() {
    
    cat <<EOF >> /etc/crontab

# post-install config

@reboot root echo heartbeat > /sys/class/leds/led0/trigger
EOF

}

update_modules() {

    cat <<EOF >> /etc/modules

# post-install config

bcm2708-rng
EOF

}

reboot() {
    shutdown -r now
}


## do stuff

update_crontab
update_modules
reboot
