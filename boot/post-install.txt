#!/bin/sh

update_crontab() {
    
    cat <<EOF >> /etc/crontab

# post-install config

@reboot root echo heartbeat > /sys/class/leds/led0/trigger
EOF

}

update_modules() {

    cat <<EOF >> /etc/modules

# post-install config

bcm2708-rng
EOF

}

# stop read heartbeat
echo none > /sys/class/leds/led1/trigger

# do stuff
update_crontab
update_modules
shutdown -r now
